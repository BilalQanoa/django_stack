from models import User

# Query: Create 3 new users

User.objects.create(first_name="Bilal", last_name="Qanoa", email="bilal@gmail.com", age=24)
User.objects.create(first_name="Khaled", last_name="Qanoa", email="bilal@gmail.com", age=23)
User.objects.create(first_name="Khaled", last_name="Abadla", email="khaled@gmail.com", age=23)

# Query: Retrieve all the users

User.objects.all()

# Query: Retrieve the last user

User.objects.last()

# Query: Retrieve the first user

User.objects.first()

# Query: Change the user with id=3 so their last name is Pancakes.

user = User.objects.get(id=3)
user.last_name = "Pancakes"
user.save()

# Query: Delete the user with id=2 from the database

User.objects.get(id=2).delete()

# Query: Get all the users, sorted by their first name

User.objects.order_by('first_name')

# BONUS Query: Get all the users, sorted by their first name in descending order

User.objects.order_by('-first_name')
